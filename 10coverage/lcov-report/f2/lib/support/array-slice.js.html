module.exports = function arraySlice(array, indexArg, endArg) {
    var index = indexArg || 0;
    var i = -1;
    var n = array.length - index;
 
    if (n < 0) {
        n = 0;
    }
 
    if (endArg > 0 && n > endArg) {
        n = endArg;
    }
 
    var array2 = new Array(n);
    while (++i < n) {
        array2[i] = array[i + index];
    }
    return array2;
};
